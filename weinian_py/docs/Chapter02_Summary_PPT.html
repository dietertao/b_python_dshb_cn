# 添加异常值和缺失值 sales += np.random.randint(0,3000,n_samples) * 0.15 sales[np.random.choice(n_samples, size=int(n_samples*0.05), replace=False)] = np.nan # 缺失值处理 mask = np.random.rand(n_samples) < 0.05 sales = np.where(mask, np.nan, sales) sales = np.round(np.where(np.isnan(sales), np.nanmean(sales), sales), decimals=2)

项目1: 统计分析函数

自定义describe_array函数

🔍 函数功能设计：

  • 基本信息： 大小、数据类型、形状、内存使用
  • 数据质量： 缺失值数量、有效值数量
  • 统计指标： 均值、中值、最值、极差
  • 分布特征： 标准差、方差、偏度、峰度
def describe_array(arr): """计算数组的基本统计信息""" if not isinstance(arr, np.ndarray): raise ValueError("Input must be a numpy array.") summary = {} summary["Size"] = arr.size summary['dtype'] = arr.dtype summary['Shape'] = arr.shape summary['NaN_count'] = np.isnan(arr).sum() summary['Mean'] = np.nanmean(arr) summary['Std'] = np.nanstd(arr) # ... 更多统计指标 return summary

💡 技术亮点：

  • 处理包含NaN值的数组（使用np.nan*函数）
  • 完整的统计描述（包括偏度和峰度）
  • 错误处理和输入验证

工具库开发 (pybds_lib)

模块化编程最佳实践

🛠️ describe_array函数特性：

  • 输入验证： 类型检查和错误处理
  • 全面统计： 15+个统计指标
  • NaN处理： 专门处理缺失值的统计函数
  • 内存分析： 数组内存使用情况
def describe_array(arr): """计算Numpy数组的基本统计信息""" summary = {} # 基本信息 summary["Size"] = arr.size summary['dtype'] = arr.dtype summary['Shape'] = arr.shape summary['Memory_usage'] = arr.nbytes # 数据质量 summary['NaN_count'] = np.isnan(arr).sum() summary['finite_count'] = np.isfinite(arr).sum() # 统计指标 (使用nan安全函数) summary['Mean'] = np.nanmean(arr) summary['Median'] = np.nanmedian(arr) summary['Std'] = np.nanstd(arr) summary['Skewness'] = np.nanmean((arr-np.nanmean(arr))**3 / (np.nanstd(arr)**3)) return summary

🎯 设计理念：

  • 可重用性：跨项目通用的统计分析工具
  • 健壮性：完善的错误处理和边界情况处理
  • 扩展性：易于添加新的统计指标
